#Preparando  el ambiente y limpieza de datos usando Stacks

#Instalación del programa

$tar xfvz stacks-2.xx.tar.gz
$cd stacks-2.xx
$./configure
$make

(become root)
% make install

(or, use sudo)
$sudo make install

#Para cambiar el directorio de instalación
$./conigue --prefix=/home/usr/local
